import matplotlib.pyplot as plt

# Bar chart of cumulative returns
plt.figure(figsize=(8,12))
df_summary["cum_return"].sort_values().plot.barh()
plt.title("Cumulative Return by Ticker")
plt.xlabel("Cumulative Return")
plt.tight_layout()
plt.show()

# Scatter of win_rate vs. cum_return
plt.figure()
plt.scatter(df_summary["win_rate"], df_summary["cum_return"])
plt.xlabel("Win Rate")
plt.ylabel("Cumulative Return")
plt.title("Win Rate vs. Cumulative Return")
plt.grid(True)
plt.show()
